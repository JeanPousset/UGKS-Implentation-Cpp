6// Updates (with BKG formula) distribution matrix F at time n+1 for each space node (x_i i in [[1,Nx+1]]) and each velocity v_j(j in [[-N,N]])
7void BGK::distribution_update(Matrix &F, const Matrix &φ, const double *ρ_np1, const Discretization &δ, double σ, double eta, double ε) const {
8 // precomputed coefficients to avoid useless multiplication and divisions
9 // -> the optimal way would be to compute these coef at the beginning of the solver, but it'll lose the genericity of the formulation
10 const double K = σ * δ.dt / (ε * eta);
11 const double coef = 1. / (1. + K);
12 const double δ_ratio = δ.dt / δ.dx;
13
14 // Special case for the first and last space nodes (x = 0, L) => periodic conditions
15 for (int j = 0; j < 2 * δ.N; ++j) {
16 F[0][j] = coef * (F[0][j] + K * ρ_np1[0] - δ_ratio * (φ[0][j] - φ[δ.Nx - 1][j]));
17 F[δ.Nx][j] = F[0][j]; // Same value (periodic conditions)
18 }
19 // Other nodes
20 for (int i = 1; i < δ.Nx; ++i) {
21 for (int j = 0; j < 2 * δ.N; ++j) {
22 F[i][j] = coef * (F[i][j] + K * ρ_np1[i] - δ_ratio * (φ[i][j] - φ[i-1][j]));
23 }
24 }
25}
