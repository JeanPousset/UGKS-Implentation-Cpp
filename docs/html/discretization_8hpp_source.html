46// Struct constructor
47Discretization::Discretization(double dt, double Tf, int N, double v_max, double L,
48 int Nx) : dt(dt), N(N), L(L), Nx(Nx) {
49 // Time discretization
50 const double Nt_tmp = Tf / dt;
51 if (std::abs(Nt_tmp - std::round(Nt_tmp)) > 1e-12) {
52 // check if Tf is divisible by dt (allow error because of numeric encoding of fload)
53 std::cout << "mod(" << Tf << "," << dt << ") = " << std::fmod(Tf, dt) << std::endl;
54 throw std::invalid_argument("Error in discretization : please choose a final time Tf divisible by dt");
55 }
56 Nt = static_cast<int>(Nt_tmp);
57 // Linear discretization of the (symmetric) velocity domain [-v_max, v_max]
58 // -> it's made so to optimize the number of multiplication / division performed
59 V = new double[2 * N];
60 double v = 0.;
61 const double dv = v_max / N;
62 normV2 = 0.;
63 for (int i = 0; i < N; ++i) {
64 v = i * dv;
65 V[i] = -v;
66 V[N + i] = v;
67 normV2 += v * v;
68 }
69
70 // Linear discretization of the space domain
71 X = new double[Nx+1];
72 dx = L / Nx; // Here the number of space node is Nx+1
73 for (int i = 0; i < Nx + 1; ++i) { X[i] = dx * i; }
74}
