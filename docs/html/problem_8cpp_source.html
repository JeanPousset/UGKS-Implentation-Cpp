8double compute_A(double λ, double σ, double eta, double ε, double dt) {
9 const double w = λ * σ * dt / (eta * ε);
10 return (exp(w) - 1.) / (eta * w);
11}
12
13double compute_C(double eta, double A) {
14 return 1. / eta - A;
15}
16
17double compute_D(double λ, double σ, double eta, double ε, double dt, double A, double C) {
18 const double w = λ * σ * dt / (eta * ε);
19 return ε / (σ * λ) * (C - A) + ε * exp(w) / (eta * σ * λ);
20}

55// Updates density vector ρ at time n+1 for each space node (x_i i in [[1,Nx+1]])
56void Problem::density_update(double *ρ_np1, const double *ρ_n, const double *Φ) const {
57 const double δ_ratio = δ_.dt / δ_.dx;
58 // special case for the first and last nodes (x = 0, L) with periodic boundary conditions
59 ρ_np1[0] = ρ_n[δ_.Nx] - δ_ratio * (Φ[0] - Φ[δ_.Nx - 1]);
60 ρ_np1[δ_.Nx] = ρ_np1[0];
61 for (int i = 1; i < δ_.Nx; ++i) {
62 ρ_np1[i] = ρ_n[i] - δ_ratio * (Φ[i] - Φ[i-1]);
63 }
64}

78// Solves the diffusion / transport problem with UGKS
79Matrix Problem::solve() const {
80 const Discretization &δ = δ_; // Simple reference for easier reading
81 // Dynamic allocation
82 Matrix F(δ.Nx + 1, 2 * δ.N); // Distribution at time n
83 double *Φ = new double[δ.Nx];
84 Matrix φ(δ.Nx, 2 * δ.N);
85 Matrix ρ(δ.Nt, δ.Nx + 1);
86
87 // Distribution and density initializations based on problem data
88 for (int i = 0; i < δ.Nx + 1; i++) {
89 ρ[0][i] = ρ0(δ.X[i]); // density initialization
90 for (int j = 0; j < 2 * δ.N; ++j) {
91 F[i][j] = f0(δ.X[i], δ.V[j]); // distribution initialization
92 }
93 }
94
95 // Solve for each time step
96 for (int k = 0; k < δ.Nt - 1; ++k) {
97 this->time_step(F, ρ[k + 1], Φ, φ, ρ[k]);
98
99 if (k % 1000 == 0) {
100 std::cout << "\t time step : " << k << "/" << δ.Nt << std::endl;
101 }
102 }
103
104 // Clean memory
105 delete[] Φ;
106
107 return ρ;
108 // Move constructor is called to move ρ data into the matrix returned, then the destructor
109 // is called on ρ that is now en empty matrix with a nullptr
110}
